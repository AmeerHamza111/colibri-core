137  bool process_block() {
138  int num = pptr() - pbase();
139  cstream.next_in = pbase();
140  cstream.avail_in = num;
141 
142  while (cstream.avail_in) {
143  // set the pointers to the output buffer
144  cstream.next_out = &out_buffer[0];
145  cstream.avail_out = out_buffer.size();
146 
147  // compress the data; we don't need to check for errors
148  // here because there is no possibility that they can
149  // happen here; it is explained in the libbzip2 docs.
150  BZ2_bzCompress(&cstream, BZ_RUN);
151 
152  // write the data to the underlying stream buffer
153  int out_size = out_buffer.size() - cstream.avail_out;
154  if (dest->sputn(&out_buffer[0], out_size) != out_size)
155  return false;
156  }
157 
158  pbump(-num);
159  return true;
160  }
161 
162  bool flush() {
163  bool flushed = false;
164  cstream.next_in = NULL;
165  cstream.avail_in = 0;
166 
167  do {
168  // set the pointers to the output buffer
169  cstream.next_out = &out_buffer[0];
170  cstream.avail_out = out_buffer.size();
171 
172  // here no errors should happen, too
173  flushed = BZ2_bzCompress(&cstream, BZ_FLUSH) == BZ_RUN_OK;
174 
175  // write the data to the underlying stream buffer
176  int out_size = out_buffer.size() - cstream.avail_out;
177  if (dest->sputn(&out_buffer[0], out_size) != out_size)
178  return false;
179  } while (!flushed);
180 
181  return true;
182  }
183 
184  bool finish() {
185  bool finished = false;
186  cstream.next_in = NULL;
187  cstream.avail_in = 0;
188 
189  do {
190  // set the pointers to the output buffer
191  cstream.next_out = &out_buffer[0];
192  cstream.avail_out = out_buffer.size();
193 
194  // here no errors should happen, too
195  finished = BZ2_bzCompress(&cstream, BZ_FINISH) == BZ_STREAM_END;
196 
197  // write the data to the underlying stream buffer
198  int out_size = out_buffer.size() - cstream.avail_out;
199  if (dest->sputn(&out_buffer[0], out_size) != out_size)
200  return false;
201  } while (!finished);
202 
203  return true;
204  }

206  virtual int_type overflow(int_type c) {
207  if (!traits_type::eq_int_type(c, traits_type::eof())) {
208  // put this character in the last position
209  // (this function is called when pptr() == eptr() but we
210  // have reserved one byte more in the constructor, thus
211  // *epptr() and now *pptr() point to valid positions)
212  *pptr() = c;
213  pbump(1);
214  }
215 
216  return process_block() ? traits_type::not_eof(c) : traits_type::eof();
217  }
218 
219  virtual int sync() {
220  return process_block() && flush() ? 0 : -1;
221  }

223  virtual std::streamsize xsputn(const char* p, std::streamsize num) {
224  // the number of chars copied
225  std::streamsize done = 0;
226 
227  // loop until no data is left
228  while (done < num) {
229  // get the number of chars to write in this iteration
230  // (we've got one more char than epptr thinks)
231  int block_size = std::min(num - done, epptr() - pptr() + 1);
232 
233  // write them
234  std::memcpy(pptr(), p + done, block_size);
235 
236  // update the write pointer
237  pbump(block_size);
238 
239  // process_block if necessary
240  if (pptr() >= epptr()) {
241  if (!process_block())
242  break;
243  }
244 
245  // update the yet-to-do count
246  done += block_size;
247  }
248 
249  return done;
250  }

340  cstream.bzalloc = bzalloc;
341  cstream.bzfree = bzfree;
342  cstream.opaque = opaque;
343

345  int ret = BZ2_bzCompressInit(&cstream, block_size_100K, verbosity,
346  work_factor);
347  // BZ_PARAM_ERROR won't happen here because we checked before
348  switch (ret) {
349  case BZ_OK:
350  break;
351  case BZ_CONFIG_ERROR:
352  throw std::runtime_error("libbz2 was not compiled correctly.");
353  case BZ_MEM_ERROR:
354  throw std::bad_alloc();
355  default:
356  throw std::runtime_error("Unknown error creating bz2 compressor "
357  "stream buffer.");
358  }
359  }

418  virtual int_type underflow() {
419  // calculate the new size of the putback area
420  int new_putback_num = std::min(gptr() - eback(),
421  putback_end - &buffer[0]);
422 
423  // copy the new putback data into the putback area
424  std::memcpy(putback_end - new_putback_num,
425  gptr() - new_putback_num, new_putback_num);
426 
427  // shovel data into the bzip stream until there is something
428  // in the output buffer
429  do {
430  // refill the input buffer if necessary
431  if (in_begin == in_end) {
432  std::streamsize read_num = source->sgetn(&in_buffer[0],
433  in_buffer.size());
434  if (read_num == 0) {
435  // we can't read anymore
436  return traits_type::eof();
437  }
438  in_begin = &in_buffer[0];
439  in_end = in_begin + read_num;
440  }
441 
442  // decompress the data
443  dstream.next_in = in_begin;
444  dstream.avail_in = in_end - in_begin;
445  dstream.next_out = putback_end;
446  dstream.avail_out = &*buffer.end() - putback_end;
447  int ret = BZ2_bzDecompress(&dstream);
448  switch (ret) {
449  case BZ_OK:
450  break;
451  case BZ_STREAM_END:
452  if (&*buffer.end() - putback_end == (int) dstream.avail_out)
453  return traits_type::eof();
454  break;
455  case BZ_DATA_ERROR:
456  case BZ_DATA_ERROR_MAGIC:
457  case BZ_MEM_ERROR:
458  default:
459  // TODO: handle these errors separately
460  return traits_type::eof();
461  }
462 
463  // update the input buffer pointers
464  in_begin = in_end - dstream.avail_in;
465 
466  } while (dstream.avail_out + putback_end == &*buffer.end());
467 
468  // update the stream buffer pointers
469  setg(putback_end - new_putback_num,
470  putback_end,
471  &*buffer.end() - dstream.avail_out);
472 
473  // return the next character
474  return traits_type::to_int_type(*gptr());
475  }
